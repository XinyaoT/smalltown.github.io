  }
};

  dance:function(){
    console.log(`${this.name} is dancing!`);
  }
};

  name:"Lisi",
  sex:"femaile"
}

function abs(x)
{
  if (typeof x=="number")
    if(x>0)
      return x;
    else
      return -x;
}
console.log(abs(-3));

// 方式2：匿名函数
var func=function(x){
  if (typeof x=="number")
    if(x>0)
      return x;
    else
      return -x;
}
console.log(func(-4));

// 方式3：箭头函数
var func1=(x,y) => {
  if (x>y)
    return x;
  else
    return y;
}
console.log(func1(234,567));
